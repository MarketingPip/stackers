Perfect for React, Vue, or vanilla JS applications.

import { OpenStacker } from 'https://cdn.openstacker.io/v1/game.js';

const game = new OpenStacker({
  container: '#game-target',
  difficulty: 'arcade',
  onWin: (score) => console.log('You won!', score)
});

game.start();
